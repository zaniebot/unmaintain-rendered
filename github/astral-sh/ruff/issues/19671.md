```python
  import pytest
  import pytest_asyncio
  
  
W @pytest.fixture(scope='function') # W: `scope='function'` is implied in `@pytest.fixture()`
  def foo():
      pass
  
  
  @pytest_asyncio.fixture(scope='function')
  async def bar():
      pass

```
