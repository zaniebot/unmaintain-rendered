Don't you mean, when textDocument/references is triggered on an overload's function name, ty should only return references specific to that overload?

@overload
def f() -> None: ...
@overload
def f(v: str) -> None: ...
#   ^ Case 1: Overload function name
def f(v: str = '') -> None: ...
#   ^ Case 2: Implementation function name

f()       # Only 2
f('foo')  # Both 1 and 2
