class Node:
    def __init__(self, name: str, parent: Optional["Node"] = None):
        self.name = name
        self.parent = parent

positions = [("root", (0, 0)), ("child", (1, 1))]

node = Node("child", parent=Node("root"))

if node.parent:
    pos = next(p for name, p in positions if name == node.parent.name)[1]
