from __future__  import annotations

class A[T]:
    def __init__(self, value: T) -> None:
        self.t: T = value
    def f(self, value: A[T]):
        _: A[T | int] = A(value.t)  # Object of type `A[T@A]` is not assignable to `A[T@A | int]`
