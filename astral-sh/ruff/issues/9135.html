Auto-quoting fails when multiple members of a union are quoted

from django.contrib.auth.models import AbstractBaseUser, AnonymousUser

def foo(
    self, user: AbstractBaseUser | AnonymousUser, view: 'type[CondorBaseViewSet]'

    pass
