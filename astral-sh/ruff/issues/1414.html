I got a false positive SyntaxError in case i am using the match on an object's attribute:

class A:
    var1 = 5


def false_positive_case():
    a = A()
    match a.var1:
        case 5:
            print('easy match')
        case _:
            print('sad')


def okay_case():
    a = [1, 2, 3]
    match a:
        case [v1, v2, *_]:
            print('easy matcH:', v1, v2)


if __name__ == '__main__':
    false_positive_case()
    okay_case()
