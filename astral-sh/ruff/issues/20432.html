impl Generator {
 pub const fn new(indent: &'a Indentation, line_ending: LineEnding) -> Self {
        Self {
            // Style preferences.
            indent,
            line_ending,
            quote: Quote::default(), // Added this
            // Internal state.
            buffer: String::new(),
            indent_depth: 0,
            num_newlines: 0,
            initial: true,
        }
    }

  pub fn with_quote(mut self, quote: Quote) -> Self {
    self.quote = quote;
    self
  }

}
