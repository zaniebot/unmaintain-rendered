`FBT001` triggers for dunder methods where positional arguments have to be used

Closed #2629 opened by @trag1c on 2023-02-07 16:58
Issue opened by @trag1c on 2023-02-07 16:58

Ruff version: 0.0.243

from enum import Enum

class Switch(Enum):
    CLASS = 0
    FUNCTION = 1
    IMPORT = 2
    BUILTIN = 3

class Registry:
    def __init__(self) -> None:
        self._switches = [False] * len(Switch)

    # FBT001: Boolean positional arg in function definition
    def __setitem__(self, switch: Switch, value: bool) -> None:
        self._switches[switch.value] = value

reg = Registry()
reg[Switch.CLASS] = True

value cannot be a keyword-only parameter as that triggers a TypeError:

__setitem__() takes 2 positional arguments but 3 were given
