        *cfg_path,
    ) as cf, dsp.sub_at(*cfg["path"], *fs_path) as fs:

    Dispatch(obj.cfg, run=True, sig=True) as dsp,

    dsp.sub_at(*cfg["path"], *fs_path) as fs,
