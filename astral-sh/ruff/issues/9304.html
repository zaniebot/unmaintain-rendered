Given:

from somewhere import get_cfg

def lookup_cfg(cfg_description):
    cfg = get_cfg(cfg_description)
    if cfg is not None:
        return cfg
    raise AttributeError(f"No cfg found matching {cfg_description}")

-def lookup_cfg(cfg_description):

     cfg = get_cfg(cfg_description)
     if cfg is not None:
         return cfg
