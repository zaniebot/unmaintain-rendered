a = 3; not a is None # noqa: E714 -> finds E702
a = 3; not a is None # noqa: not-is-test -> suppresses both errors

Expectations:

a = 3; not a is None # noqa: E714 -> finds E702
a = 3; not a is None # noqa: not-is-test -> finds E702
