fn hover_regression_test_366() {

        r#"

    class A: ...
    class B: ...

    def _(x: Intersection[A,<CURSOR> B]):

    7 |         def _(x: Intersection[A, B]):
      |                               ^^-^
      |                               | |

      |                               source
    8 |             pass
      |
    ");
}
