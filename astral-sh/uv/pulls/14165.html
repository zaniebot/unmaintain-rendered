          for image in $IMAGES; do
            docker buildx imagetools create \
            $(jq -cr --arg img "${image}" '.tags | map(select(startswith($img)) | "-t " + .) | join(" ")' <<< "$DOCKER_METADATA_OUTPUT_JSON") \
            "${image}@${DIGEST}"
          done
