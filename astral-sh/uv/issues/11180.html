Summary

https://github.com/SchemaStore/schemastore/pull/4224#issuecomment-2630683734

The following passes validation:

[tool.uv]
conflicts = [
    [
        { conflict = { Extra = "cpu" } },
        { conflict = { Extra = "cu124" } },
    ],
]

but UV expects:

[tool.uv]
conflicts = [
    [
        { extra = "cpu" },
        { extra = "cu124" },
    ],
]
