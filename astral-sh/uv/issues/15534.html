Summary

uv pip compile pyproject.toml does not produce a uv pip install -rable requirements.txt when extra repositories are used. This is because in pyproject.toml I'm using torch from a specific index, which is not reflected in the requirements.txt.

[project]
name = "example"
version = "0.1.0"
requires-python = ">=3.12"
dependencies = [
    "torch"
]

environments = [
    "sys_platform == 'linux' and implementation_name == 'cpython' and python_version == '3.11'",
]

# https://docs.astral.sh/uv/concepts/projects/dependencies/#dependency-sources
[tool.uv.sources]

# https://docs.astral.sh/uv/guides/integration/pytorch/#using-uv-with-pytorch
torch = [
    { index = "pytorch-cuda" },
]

# https://docs.astral.sh/uv/concepts/indexes/#package-indexes
# TODO(jdanek): explicit = false seems to be a workaround, otherwise `uv pip compile --emit-index-url` wont emit it
#  also see https://github.com/astral-sh/uv/issues/10008
[[tool.uv.index]]
name = "pytorch-cuda"
url = "https://download.pytorch.org/whl/cu126"
explicit = true

❯ uv pip compile pyproject.toml --python-platform linux  --emit-index-url  --emit-index-annotation
