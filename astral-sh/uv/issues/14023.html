e.g.

❯ cat $(uv tool dir)/ruff/uv-receipt.toml
[tool]
requirements = [
    { name = "ruff" },
    { name = "httpie" },
    { name = "httpie" },

entrypoints = [
    { name = "ruff", install-path = "/Users/zb/.local/bin/ruff", from = "ruff" },
