[`uv_build`] support `dynamic = ["version"]`

#14946 opened by @chirizxc on 2025-07-28 20:37
Issue opened by @chirizxc on 2025-07-28 20:37

Summary

[project]
dynamic = ["version"]

Will there be support for dynamic versioning, e.g. so that it is taken from GitHub tags like hatch-vcs?

[build-system]
requires = ["hatchling", "hatch-vcs"]
build-backend = "hatchling.build"

[tool.hatch.version]
source = "vcs"

[tool.hatch.version.vcs]
tag-pattern = "*v"

[project]
dynamic = ["version"]

Example

[build-system]
requires = ["uv_build>=0.8.3,<0.9.0"]
build-backend = "uv_build"

[tool.uv.build-backend.version]
source = "vcs"
tag-pattern = "*v"

[project]
dynamic = ["version"]
