#6127 opened by @charliermarsh on 2024-08-15 20:07
@charliermarsh

Given:

[project]
name = "infrastructure"
version = "0.0.0"
dependencies = []

[project.optional-dependencies]
ansible = [
    "ansible>=9.4.0",
]
lint_ansible = [
     "ansible-lint>=24.2.2; platform_system != 'Windows'",
]

[build-system]
requires = ["setuptools", "wheel"]
build-backend = "setuptools.build_meta"

We start solving ansible-lint without forking. It then hits a dependency on will-not-work-on-windows-try-from-wsl-instead{platform_system == 'Windows'}==0.1.0 (that's literally the name of the package). We don't notice that the dependency is disjoint from the requirer, because we're not in a fork, so we aren't tracking markers.
