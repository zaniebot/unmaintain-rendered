Minimal reproducible code 
use clap::Clap;

#[derive(Clap)]
enum Cmd {
    #[clap(visible_alias = "alias")]
    Test { args: Vec<String> },
}

#[derive(Clap)]
struct Cli {
    #[clap(subcommand)]
    cmd: Cmd,
}

fn main() {
    let cli = Cli::parse();
}
