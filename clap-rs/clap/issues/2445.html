Minimal reproducible code

use clap;

fn main() {
    let _args = clap::App::new("test")
        .set_term_width(80)

        .arg(clap::Arg::with_name("foo")
            .short("f")
            .long("foo")
            .takes_value(true)
            .value_name("FOO")
            .required(false)
            .long_help("This is a very very very very very very very very very very very long help message"))

        .arg(clap::Arg::with_name("bar")
            .short("b")
            .long("bar")
            .takes_value(true)
            .value_name("BAR")
            .required(false)
            .long_help("This is a very very very very very very very very very very very long help message"))

        .arg(clap::Arg::with_name("baz")
            .short("z")
            .long("baz")
            .takes_value(true)
            .value_name("BAZ")
            .required(false)
            .long_help("This is a short message"))

        .arg(clap::Arg::with_name("qux")
            .short("q")
            .long("qux")
            .takes_value(true)
            .value_name("QUX")
            .required(false)
            .long_help("This is a very very very very very very very very very very very long help message"))

        .arg(clap::Arg::with_name("quux")
            .short("u")
            .long("quux")
            .takes_value(true)
            .value_name("QUUX")
            .required(false)
            .long_help("This is a very very very very very very very very very very very long help message"))

        .get_matches();
}
