Minimal reproducible code

/*
[dependencies]
clap = { version = "4.3.16", features = ["derive"] }
*/

use clap::Parser;

pub type MountOpt = Vec<String>;

#[derive(Clone, Debug, Parser)]
pub struct Args {
    #[arg(long, value_name = "mount")]
    pub mount: Option<MountOpt>,
}

fn main() {
    println!("{:?}", Args::parse());
}
