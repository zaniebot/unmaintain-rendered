Clap Version

4.4.17

Minimal reproducible code

use clap::Parser;
use clap::Subcommand;

#[derive(Parser)]
struct Args {
    #[command(subcommand)]
    subcommand: Sub,
}

#[derive(Parser)]
struct ExpireSub {
    filename: String,
    token: String,
    expires: u64,
}

#[derive(Parser)]
struct DeleteSub {
    filename: String,
    token: String,
}

#[derive(Subcommand)]
enum Sub {
    Expire(ExpireSub),
    Delete(DeleteSub),
}

fn main() {
    let _args = Args::parse();
}

I also tried using Args instead of Parser for subcommands but result is the same.

Steps to reproduce the bug with the above code

cargo run -- delete
cargo run -- expire
cargo run -- expire 1
