Minimal reproducible code

use clap::{Parser, Subcommand};

#[derive(Parser, Debug)]
struct Opt {
    #[clap(subcommand)]
    cmd: Option<Cmds>,
}

#[derive(Subcommand, Debug)]
enum Cmds {
    #[clap(subcommand)]
    One(OneCmd),
}

#[derive(Subcommand, Debug)]
enum OneCmd {
    Add { name: String, items: Vec<String> },
}

fn main() {
    Opt::parse();
}
