#[derive(Parser)]
struct Main {
  #[clap(subcommand)]

}

  Test(Sub),
}

#[derive(Parser)]

}

fn main() {
  clap_complete::Shell::Bash.generate(&Main::into_app()), &mut io::stdout());
}
