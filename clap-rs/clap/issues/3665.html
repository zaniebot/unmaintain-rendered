use clap::{Command, Arg};

fn main() {
    let _matches = Command::new("Clap-created-USAGE-string-bug")
        .arg(
            Arg::new("a").required(true)
        )
        .arg(
            Arg::new("b")
                .short('b')
                .takes_value(true)
                .conflicts_with("c")
                // .group("debug")
        )
        .arg(
            Arg::new("c")
                .short('c')
                .takes_value(true)
                .conflicts_with("b")
                // .group("debug")
        )

}
