use std::io;
use clap::{Parser, CommandFactory};
use clap_complete::generate;
use clap_complete_nushell::Nushell;

fn main() {
    let mut cmd = Cli::command_for_update();
    generate(Nushell, &mut cmd, "foo", &mut io::stdout());
}

#[derive(Parser)]
struct Cli {
    /// Multiline
    /// comment.
    #[arg(long, verbatim_doc_comment)]
