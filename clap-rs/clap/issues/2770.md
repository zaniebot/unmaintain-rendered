```rust
use clap::{App, Arg};

fn main() {
    let _matches = App::new("Clap bug")
        .arg(
            Arg::new("a")
                .short('a')
                .required_unless_present_any(&["b", "c"])
                .conflicts_with_all(&["b", "c"]),
        )
        .arg(
            Arg::new("b")
                .short('b')
                .required_unless_present("a")
                .requires("c"),
        )
        .arg(
            Arg::new("c")
                .short('c')
                .required_unless_present("a")
                .requires("b"),
        )
        .get_matches();
}
```
