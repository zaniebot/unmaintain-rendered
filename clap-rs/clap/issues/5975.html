Clap Version

4.5.37

Minimal reproducible code

use clap::{Args, CommandFactory, Parser, Subcommand};
use clap_complete::{generate, Shell};

#[derive(Parser)]
#[command(name = "clap_test", author, version, about)]
pub struct Cli {
    #[arg(long = "generate", hide = true, value_enum)]
    pub generator: Option<Shell>,

    #[command(subcommand)]
    pub command: Option<Commands>,
}

#[derive(Subcommand)]
pub enum Commands {
    First(FirstCommand),
}

#[derive(Args)]
pub struct FirstCommand {
    #[command(subcommand)]
    pub command: SecondCommand,
}

#[derive(Subcommand)]
pub enum SecondCommand {
    Second(SecondCommandArgs),
}

#[derive(Args)]
pub struct SecondCommandArgs {
    #[arg(long, short)]
    pub flag: bool,

    #[command(subcommand)]
    pub command: ThirdCommand,
}

#[derive(Subcommand)]
pub enum ThirdCommand {
    Third,
}

fn main() {
    let cli = Cli::parse();

    if let Some(generator) = cli.generator {
        let mut cmd = Cli::command();
        eprintln!("Generating completion file for {generator:?}...");
        let cmd = &mut cmd;
        generate(
            generator,
            cmd,
            cmd.get_name().to_string(),
            &mut std::io::stdout(),
        );
        return;
    }

    if let Some(Commands::First(first_command)) = cli.command {
        match first_command.command {
            SecondCommand::Second(second_command) => {
                match second_command.command {
                    ThirdCommand::Third {} => {
                        println!("Third command called");
                    }
                }
            }
        }
    }
}

Steps to reproduce the bug with the above code

# On the fish shell : 
# Build the CLI
cargo build
# Generate fish completions
cargo run -- --generate=fish > completions.fish
# Source fish completions
source completions.fish
# Trigger the completion on the third subcommand
./target/debug/clap_test first second third <TAB>

Actual Behaviour

The fish shell display the following completion :

❯ ./target/debug/clap_test first second third 
help  (Print this message or the help of the given subcommand(s))  third

Expected Behaviour

The fish shell should display the following completion :

❯ ./target/debug/clap_test first second third 
Cargo.lock  Cargo.toml  completions.fish  src/  target/

Additional Context

This same issue also occurs with arguments.

In this example there is a flag argument on the second command, the completion of the third command will display it as well :

This issue is not present with the generated Bash completion.
