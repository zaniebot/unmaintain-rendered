Minimal reproducible code 
use clap::Parser;

#[derive(Parser)]
#[command(version)]
struct Cli {
    #[arg(long, default_value_t = 0)]
    small: u8,
    #[arg(long, default_value_t = false)]
    debug: bool,
}

fn main() {
    let _args = Cli::parse();
}

$ cargo run -q -- --help

Usage: default-values [OPTIONS]
