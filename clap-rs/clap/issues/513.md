pub fn make_app<'a, 'b: 'a>() -> App<'a, 'b> {

        .arg(Arg::with_name("command")
             .value_name("command")
             .help("Command to execute"))
        .arg(Arg::with_name("args")
             .value_name("args")

             .required(false)

}
