rg  --multiline --only-matching '^(?P<indent>\s*)a=(?P<value>(?ms:[(].*?[)])|.*?)$' --replace '${value}' ./test.bash

rg  --multiline --only-matching '^(?-ms:(?P<indent>\s*)a=(?P<value>[(](?ms:.*?)[)]|[^(].*?))$' --replace '${value}' test.bash
